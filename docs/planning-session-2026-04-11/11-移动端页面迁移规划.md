# 移动端页面迁移规划

> 从页面功能出发，反向推导数据和后端需求

---

## 一、迁移策略

### 核心原则

```
┌────────────────────────────────────────────────────────────┐
│                    迁移顺序                                  │
├────────────────────────────────────────────────────────────┤
│                                                            │
│   1. 分析页面功能 → 2. 确定数据需求 → 3. 迁移数据表          │
│                                                            │
│        ↓                        ↓                 ↓        │
│                                                            │
│   4. 迁移后端API → 5. 迁移WatermelonDB Model → 6. 迁移页面  │
│                                                            │
│        ↓                 ↓                    ↓            │
│                                                            │
│   7. 实现 synchronize() → 8. 验证离线流程                   │
│                                                            │
└────────────────────────────────────────────────────────────┘
```

### 每个页面的迁移流程

```
单个页面迁移步骤：

Step 1: 分析页面功能
  - 读取源页面代码
  - 理解业务逻辑
  - 列出依赖的 Store、API、组件

Step 2: 确定数据需求
  - 数据表：需要哪些 Drizzle schema
  - WatermelonDB Model：需要哪些本地 Model
  - API endpoint：需要哪些 tRPC router

Step 3: 迁移数据表（后端 Drizzle）
  - 定义 schema
  - 生成 migration
  - 验证数据库结构

Step 4: 迁移后端 API
  - 实现 tRPC router
  - 实现相关 procedures
  - 测试 endpoint

Step 5: 迁移 WatermelonDB Model
  - 定义 Model（@text, @date 等）
  - 定义 Schema
  - 定义 Migration

Step 6: 实现 sync endpoint
  - sync.pull：返回增量数据
  - sync.push：接收客户端变更

Step 7: 迁移页面代码
  - 替换 Zustand → useObservation
  - 替换 API 调用 → synchronize()
  - 替换手动刷新 → 自动订阅

Step 8: 验证离线流程
  - 离线创建 → 网络恢复 → 同步成功
  - 冲突处理测试
  - UI 自动刷新验证
```

// apps/mobile/lib/monitoring/config.ts

export const MONITORING_ENVIRONMENT = 
  process.env.EXPO_PUBLIC_MONITORING_ENVIRONMENT || 
  (__DEV__ ? 'development' : 'production');

export const MONITORING_RELEASE = `${APP_SLUG}@${APP_VERSION}`;

export const SENTRY_DSN = process.env.EXPO_PUBLIC_SENTRY_DSN;
export const SENTRY_ENABLED = Boolean(SENTRY_DSN);

export const POSTHOG_API_KEY = process.env.EXPO_PUBLIC_POSTHOG_KEY;
export const POSTHOG_HOST = process.env.EXPO_PUBLIC_POSTHOG_HOST || 'https://us.i.posthog.com';
export const POSTHOG_ENABLED = Boolean(POSTHOG_API_KEY);

// Step 3: Sentry 初始化
// apps/mobile/lib/monitoring/sentry.ts

import * as Sentry from '@sentry/react-native';

export const sentryNavigationIntegration = Sentry.reactNavigationIntegration({
  enableTimeToInitialDisplay: !__DEV__,
});

export function initializeSentry() {
  if (!SENTRY_ENABLED) return;
  
  Sentry.init({
    dsn: SENTRY_DSN,
    environment: MONITORING_ENVIRONMENT,
    release: MONITORING_RELEASE,
    tracesSampleRate: 0.2,
    integrations: [sentryNavigationIntegration],
    enableNativeFramesTracking: !__DEV__,
  });
  
  Sentry.setTag('app.environment', MONITORING_ENVIRONMENT);
  Sentry.setTag('app.platform', 'mobile');
}

export function setSentryUser(user: { id: string }) {
  Sentry.setUser({ id: user.id });
}

export function clearSentryUser() {
  Sentry.setUser(null);
}

// Step 4: PostHog 初始化
// apps/mobile/lib/monitoring/posthog.ts

import PostHog from 'posthog-react-native';

export const posthogClient = POSTHOG_ENABLED 
  ? new PostHog(POSTHOG_API_KEY, {
      host: POSTHOG_HOST,
      persistence: 'file',
      captureAppLifecycleEvents: true,
    })
  : null;

export function captureAnalyticsEvent(event: string, properties?: object) {
  if (!posthogClient) return;
  posthogClient.capture(event, properties);
}

export function captureScreenView(pathname: string) {
  captureAnalyticsEvent('$pageview', { pathname });
}

export function identifyAnalyticsUser(user: { id: string; name: string }) {
  if (!posthogClient) return;
  posthogClient.identify(user.id, { name: user.name });
}

export function resetAnalyticsUser() {
  if (!posthogClient) return;
  posthogClient.reset();
}

#### Step 4: sync endpoint

```typescript
// apps/backend/src/modules/sync/sync.router.ts

@defineRouter()
export class SyncRouter {
  constructor(private projectService: ProjectsService) {}
  
  @procedure.input(z.object({ lastPulledAt: z.number() }))
  async pull({ lastPulledAt }: { lastPulledAt: number }) {
    const timestamp = Date.now();
    
    const projects = await this.projectService.findUpdatedSince(lastPulledAt);
    
    return {
      changes: {
        projects: {
          created: projects.filter(p => p.createdAt > lastPulledAt),
          updated: projects.filter(p => p.updatedAt > lastPulledAt && p.createdAt <= lastPulledAt),
          deleted: [], // 从 tombstones 获取
        }
      },
      timestamp
    };
  }
  
  @procedure.input(z.object({ changes: z.any() }))
  async push({ changes }: { changes: any }) {
    if (changes.projects) {
      for (const created of changes.projects.created) {
        await this.projectService.create(created);
      }
      for (const updated of changes.projects.updated) {
        await this.projectService.update(updated._id, updated);
      }
      for (const deletedId of changes.projects.deleted) {
        await this.projectService.delete(deletedId);
      }
    }
  }
}
```
